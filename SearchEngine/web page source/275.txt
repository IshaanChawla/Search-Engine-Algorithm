   var total_groups = 3; 


$('#btnLoadMore').click(function(){
	if(track_load <= total_groups && loading==false) 
	{
        loading = true; 
		$('.animation_image').show();   
		$('#divLoadMore').hide();
		$.post('absoluteclasses/ajax/auto_load_course.php',{'group_no': track_load,'ty':'Full-All'}, function(data){		
			if(data != '')
			{
			   setTimeout(function(){
				  $('.animation_image').hide();			
				  $('#divLoadMore').show();
			   },1800);
		   } else {
				 $('.animation_image').hide();			
				  $('#divLoadMore').hide();
		   }
		   setTimeout(function(){
			  $("#results").append(data);								
			  track_load++; 
			  loading = false; 
		   },2000); 	
		}).fail(function(xhr, ajaxOptions, thrownError) {              
		   $('.animation_image').hide(); 
			$('#divLoadMore').hide();
		   loading = false;					
		});
	} else {
		 $('.animation_image').hide(); 
		$('#divLoadMore').hide();
		 // loading = false;				
	}
});
   
});
