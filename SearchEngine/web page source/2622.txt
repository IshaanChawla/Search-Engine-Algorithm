                     <div>
                        <h3 class="title"><a name="piechart_project"></a>3.1.&nbsp;Create Project
                        </h3>
                     </div>
                  </div>
               </div>
               <p>
                  Create a new Java project "de.vogella.jfreechart.swing.pie".
                  Create also a package "de.vogella.jfreechart.swing.pie".
                  
               </p>
            </div>
            <div class="section">
               <div class="titlepage">
                  <div>
                     <div>
                        <h3 class="title"><a name="piechart_jars"></a>3.2.&nbsp;Add jars to build path of your project
                        </h3>
                     </div>
                  </div>
               </div>
               <p> In your project create a folder "lib", and paste the JFreeChart
                  jars into this folder. 
               </p>
               <p>
                  You only need the following libraries. Please replace the version
                  numbers with the
                  version you are using.
                  
                  
               </p>
               <div class="itemizedlist">
                  <ul class="itemizedlist" type="disc">
                     <li class="listitem">
                        <p>jcommon-1.0.16.jar</p>
                     </li>
                     <li class="listitem">
                        <p>jfreechart-1.0.13.jar</p>
                     </li>
                  </ul>
               </div>
               <p>
                  
                  
               </p>
               <p>
                  Add these libraries to your classpath. See
                  <a class="ulink" href="http://www.vogella.com/tutorials/Eclipse/article.html#classpath" target="_top">
                     Eclipse Java IDE - Classpath</a>
                  for details).
                  
               </p>
            </div>

               <p>Create the following two classes. </p>
               <p>
                  
               </p><pre class="programlisting"><span class="hl-keyword">package</span> de.vogella.jfreechart.swing.pie;

<span class="hl-keyword">import</span> javax.swing.JFrame;

<span class="hl-keyword">import</span> org.jfree.chart.ChartFactory;
<span class="hl-keyword">import</span> org.jfree.chart.ChartPanel;
<span class="hl-keyword">import</span> org.jfree.chart.JFreeChart;
<span class="hl-keyword">import</span> org.jfree.chart.plot.PiePlot3D;
<span class="hl-keyword">import</span> org.jfree.data.general.DefaultPieDataset;
<span class="hl-keyword">import</span> org.jfree.data.general.PieDataset;
<span class="hl-keyword">import</span> org.jfree.util.Rotation;
<span class="hl-keyword">package</span> de.vogella.jfreechart.swing.pie;


<span class="hl-keyword">public</span> <span class="hl-keyword">class</span> PieChart <span class="hl-keyword">extends</span> JFrame {

  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> <span class="hl-keyword">final</span> <span class="hl-keyword">long</span> serialVersionUID = <span class="hl-number">1L</span>;

  <span class="hl-keyword">public</span> PieChart(String applicationTitle, String chartTitle) {
        <span class="hl-keyword">super</span>(applicationTitle);
        <em class="hl-comment" style="color: #080">// This will create the dataset </em>
        PieDataset dataset = createDataset();
        <em class="hl-comment" style="color: #080">// based on the dataset we create the chart</em>
        JFreeChart chart = createChart(dataset, chartTitle);
        <em class="hl-comment" style="color: #080">// we put the chart into a panel</em>
        ChartPanel chartPanel = <span class="hl-keyword">new</span> ChartPanel(chart);
        <em class="hl-comment" style="color: #080">// default size</em>
        chartPanel.setPreferredSize(<span class="hl-keyword">new</span> java.awt.Dimension(<span class="hl-number">500</span>, <span class="hl-number">270</span>));
        <em class="hl-comment" style="color: #080">// add it to our application</em>
        setContentPane(chartPanel);

    }
    
    <div class="hl-tag" style="color: #080">/**
     * Creates a sample dataset 
     */</div>
    <span class="hl-keyword">private</span>  PieDataset createDataset() {
        DefaultPieDataset result = <span class="hl-keyword">new</span> DefaultPieDataset();
        result.setValue(<span class="hl-string" style="color:blue">"Linux"</span>, <span class="hl-number">29</span>);
        result.setValue(<span class="hl-string" style="color:blue">"Mac"</span>, <span class="hl-number">20</span>);
        result.setValue(<span class="hl-string" style="color:blue">"Windows"</span>, <span class="hl-number">51</span>);
        <span class="hl-keyword">return</span> result;
        
    }
    
    <div class="hl-tag" style="color: #080">/**
     * Creates a chart
     */</div>
    <span class="hl-keyword">private</span> JFreeChart createChart(PieDataset dataset, String title) {
        
        JFreeChart chart = ChartFactory.createPieChart3D(title,          <em class="hl-comment" style="color: #080">// chart title</em>
            dataset,                <em class="hl-comment" style="color: #080">// data</em>
            true,                   <em class="hl-comment" style="color: #080">// include legend</em>
            true,
            false);

        PiePlot3D plot = (PiePlot3D) chart.getPlot();
        plot.setStartAngle(<span class="hl-number">290</span>);
        plot.setDirection(Rotation.CLOCKWISE);
        plot.setForegroundAlpha(<span class="hl-number">0.5f</span>);
        <span class="hl-keyword">return</span> chart;
        
    }
} </pre><p>
