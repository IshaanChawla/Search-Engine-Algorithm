<p>This Java program implements the following Multi Level Inheritance:</p>
<code><pre>
Class: Account 
Cust_name , acc_no
                          
Class: Saving_Acc
Min_bal, saving_bal

Class:Acct_Details
Deposits, withdrawals
</pre><code>
<p>Program</p>
<code><pre>
import java.lang.*;
import java.io.*;
class Account
{
	String cust_name;
	int acc_no;
	Account(String a, int b)
	{
		cust_name=a;
		acc_no=b;
	}
	void display()
	{
		System.out.println (&quot;Customer Name: &quot;+cust_name);
		System.out.println (&quot;Account No: &quot;+acc_no);
	}
}
class Saving_Acc extends Account
{
	int min_bal,saving_bal;
	Saving_Acc(String a, int b, int c, int d)
	{
		super(a,b);
		min_bal=c;
		saving_bal=d;
	}
	void display()
	{
		super.display();
		System.out.println (&quot;Minimum Balance: &quot;+min_bal);
		System.out.println (&quot;Saving Balance: &quot;+saving_bal);
	}
}
class Acct_Details extends Saving_Acc
{
	int deposits, withdrawals;
	Acct_Details(String a, int b, int c, int d, int e, int f)
	{
		super(a,b,c,d);
		deposits=e;
		withdrawals=f;
	}
	void display()
	{
		super.display();
		System.out.println (&quot;Deposit: &quot;+deposits);
		System.out.println (&quot;Withdrawals: &quot;+withdrawals);
	}
}
class q9Multilevel
{
	public static void main(String args[])
	{
		Acct_Details A = new Acct_Details(&quot;Pa.one&quot;,666,1000,5000,500,9000);
		A.display();
	}
}</pre></code><br>
