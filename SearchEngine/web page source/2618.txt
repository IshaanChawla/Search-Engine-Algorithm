                  <div>
                     <h2 class="title"><a name="itext"></a>1.&nbsp;Overview
                     </h2>
                  </div>
               </div>
            </div>
            <p>
               iText is a Java library originally created by Bruno Lowagie which
               allows to create PDF, read PDF and manipulate them.
               The following
               tutorial will show how to create PDF files with
               iText. This tutorial
               assumes that you have basis
               <a class="ulink" href="http://www.vogella.com/tutorials/JavaIntroduction/article.html" target="_top"> Java</a>
               and
               <a class="ulink" href="http://www.vogella.com/tutorials/Eclipse/article.html" target="_top"> Eclipse</a>
               knowledge.
               
            </p>
            <p>iText has a hierarchical structure. The smallest text unit is a
               "Chunk" which is a String with a pre-defined font. A "Phrase" combines
               several Chunks and allows to define line spacing. "Paragraph" is a
               subclass of "Phrase" and allows to define more layout attributes, e.g.
               margins. The class "Anchor" is a subclass of "Paragraph" and serves as
               the basis for hyperlinks in the generated PDF.
               
            </p>
         </div>

                  <div>
                     <h2 class="title"><a name="createpdf"></a>3.&nbsp;Create a PDF
                     </h2>
                  </div>
               </div>
            </div>
            <p>
               Create a new Java project "de.vogella.itext.write" with the package
               "de.vogella.itext.write".
               Create a folder
               "lib" and put the iText library
               (jar file) into
               this
               folder.
               <a class="ulink" href="http://www.vogella.com/tutorials/Eclipse/article.html#classpath" target="_top">
                  Add the jar to your classpath.</a>
               
            </p>
            <p>
               Create the following class "FirstPdf.java" .
               I assume that the
               code is
               pretty much self-explaining. I tried
               to add lots of comments
               to make it
               easier to understand. For more
               complex examples have a look
               at the
               <a class="ulink" href="http://www.lowagie.com/iText/" target="_top">
                  iText Homepage</a>.
               
            </p>
            <p>
               
            </p><pre class="programlisting"><span class="hl-keyword">package</span> de.vogella.itext.write;

<span class="hl-keyword">import</span> com.itextpdf.text.Document;
<span class="hl-keyword">import</span> com.itextpdf.text.DocumentException;
<span class="hl-keyword">import</span> com.itextpdf.text.Element;
<span class="hl-keyword">import</span> com.itextpdf.text.Font;
<span class="hl-keyword">import</span> com.itextpdf.text.List;
<span class="hl-keyword">import</span> com.itextpdf.text.ListItem;
<span class="hl-keyword">import</span> com.itextpdf.text.Paragraph;
<span class="hl-keyword">import</span> com.itextpdf.text.Phrase;
<span class="hl-keyword">import</span> com.itextpdf.text.Section;
<span class="hl-keyword">import</span> com.itextpdf.text.pdf.PdfPCell;
<span class="hl-keyword">import</span> com.itextpdf.text.pdf.PdfPTable;
<span class="hl-keyword">import</span> com.itextpdf.text.pdf.PdfWriter;


<span class="hl-keyword">public</span> <span class="hl-keyword">class</span> FirstPdf {
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> String FILE = <span class="hl-string" style="color:blue">"c:/temp/FirstPdf.pdf"</span>;
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> Font catFont = <span class="hl-keyword">new</span> Font(Font.FontFamily.TIMES_ROMAN, <span class="hl-number">18</span>,
      Font.BOLD);
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> Font redFont = <span class="hl-keyword">new</span> Font(Font.FontFamily.TIMES_ROMAN, <span class="hl-number">12</span>,
      Font.NORMAL, BaseColor.RED);
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> Font subFont = <span class="hl-keyword">new</span> Font(Font.FontFamily.TIMES_ROMAN, <span class="hl-number">16</span>,
      Font.BOLD);
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> Font smallBold = <span class="hl-keyword">new</span> Font(Font.FontFamily.TIMES_ROMAN, <span class="hl-number">12</span>,
      Font.BOLD);

  <span class="hl-keyword">public</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> main(String[] args) {
    <span class="hl-keyword">try</span> {
      Document document = <span class="hl-keyword">new</span> Document();
      PdfWriter.getInstance(document, <span class="hl-keyword">new</span> FileOutputStream(FILE));
      document.open();
      addMetaData(document);
      addTitlePage(document);
      addContent(document);
      document.close();
    } <span class="hl-keyword">catch</span> (Exception e) {
      e.printStackTrace();
    }
  }

  <em class="hl-comment" style="color: #080">// iText allows to add metadata to the PDF which can be viewed in your Adobe</em>
  <em class="hl-comment" style="color: #080">// Reader</em>
  <em class="hl-comment" style="color: #080">// under File -&gt; Properties</em>
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> addMetaData(Document document) {
    document.addTitle(<span class="hl-string" style="color:blue">"My first PDF"</span>);
    document.addSubject(<span class="hl-string" style="color:blue">"Using iText"</span>);
    document.addKeywords(<span class="hl-string" style="color:blue">"Java, PDF, iText"</span>);
    document.addAuthor(<span class="hl-string" style="color:blue">"Lars Vogel"</span>);
    document.addCreator(<span class="hl-string" style="color:blue">"Lars Vogel"</span>);
  }

  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> addTitlePage(Document document)
      <span class="hl-keyword">throws</span> DocumentException {
    Paragraph preface = <span class="hl-keyword">new</span> Paragraph();
    <em class="hl-comment" style="color: #080">// We add one empty line</em>
    addEmptyLine(preface, <span class="hl-number">1</span>);
    <em class="hl-comment" style="color: #080">// Lets write a big header</em>
    preface.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Title of the document"</span>, catFont));

    addEmptyLine(preface, <span class="hl-number">1</span>);
    <em class="hl-comment" style="color: #080">// Will create: Report generated by: _name, _date</em>
    preface.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Report generated by: "</span> + System.getProperty(<span class="hl-string" style="color:blue">"user.name"</span>) + <span class="hl-string" style="color:blue">", "</span> + <span class="hl-keyword">new</span> Date(), <em class="hl-comment" style="color: #080">//$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$</em>
        smallBold));
    addEmptyLine(preface, <span class="hl-number">3</span>);
    preface.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"This document describes something which is very important "</span>,
        smallBold));

    addEmptyLine(preface, <span class="hl-number">8</span>);

    preface.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"This document is a preliminary version and not subject to your license agreement or any other agreement with vogella.com ;-)."</span>,
        redFont));

    document.add(preface);
    <em class="hl-comment" style="color: #080">// Start a new page</em>
    document.newPage();
  }

  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> addContent(Document document) <span class="hl-keyword">throws</span> DocumentException {
    Anchor anchor = <span class="hl-keyword">new</span> Anchor(<span class="hl-string" style="color:blue">"First Chapter"</span>, catFont);
    anchor.setName(<span class="hl-string" style="color:blue">"First Chapter"</span>);

    <em class="hl-comment" style="color: #080">// Second parameter is the number of the chapter</em>
    Chapter catPart = <span class="hl-keyword">new</span> Chapter(<span class="hl-keyword">new</span> Paragraph(anchor), <span class="hl-number">1</span>);

    Paragraph subPara = <span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Subcategory 1"</span>, subFont);
    Section subCatPart = catPart.addSection(subPara);
    subCatPart.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Hello"</span>));

    subPara = <span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Subcategory 2"</span>, subFont);
    subCatPart = catPart.addSection(subPara);
    subCatPart.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Paragraph 1"</span>));
    subCatPart.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Paragraph 2"</span>));
    subCatPart.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Paragraph 3"</span>));

    <em class="hl-comment" style="color: #080">// add a list</em>
    createList(subCatPart);
    Paragraph paragraph = <span class="hl-keyword">new</span> Paragraph();
    addEmptyLine(paragraph, <span class="hl-number">5</span>);
    subCatPart.add(paragraph);

    <em class="hl-comment" style="color: #080">// add a table</em>
    createTable(subCatPart);

    <em class="hl-comment" style="color: #080">// now add all this to the document</em>
    document.add(catPart);

    <em class="hl-comment" style="color: #080">// Next section</em>
    anchor = <span class="hl-keyword">new</span> Anchor(<span class="hl-string" style="color:blue">"Second Chapter"</span>, catFont);
    anchor.setName(<span class="hl-string" style="color:blue">"Second Chapter"</span>);

    <em class="hl-comment" style="color: #080">// Second parameter is the number of the chapter</em>
    catPart = <span class="hl-keyword">new</span> Chapter(<span class="hl-keyword">new</span> Paragraph(anchor), <span class="hl-number">1</span>);

    subPara = <span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"Subcategory"</span>, subFont);
    subCatPart = catPart.addSection(subPara);
    subCatPart.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">"This is a very important message"</span>));

    <em class="hl-comment" style="color: #080">// now add all this to the document</em>
    document.add(catPart);

  }

  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> createTable(Section subCatPart)
      <span class="hl-keyword">throws</span> BadElementException {
    PdfPTable table = <span class="hl-keyword">new</span> PdfPTable(<span class="hl-number">3</span>);

    <em class="hl-comment" style="color: #080">// t.setBorderColor(BaseColor.GRAY);</em>
    <em class="hl-comment" style="color: #080">// t.setPadding(4);</em>
    <em class="hl-comment" style="color: #080">// t.setSpacing(4);</em>
    <em class="hl-comment" style="color: #080">// t.setBorderWidth(1);</em>

    PdfPCell c1 = <span class="hl-keyword">new</span> PdfPCell(<span class="hl-keyword">new</span> Phrase(<span class="hl-string" style="color:blue">"Table Header 1"</span>));
    c1.setHorizontalAlignment(Element.ALIGN_CENTER);
    table.addCell(c1);

    c1 = <span class="hl-keyword">new</span> PdfPCell(<span class="hl-keyword">new</span> Phrase(<span class="hl-string" style="color:blue">"Table Header 2"</span>));
    c1.setHorizontalAlignment(Element.ALIGN_CENTER);
    table.addCell(c1);

    c1 = <span class="hl-keyword">new</span> PdfPCell(<span class="hl-keyword">new</span> Phrase(<span class="hl-string" style="color:blue">"Table Header 3"</span>));
    c1.setHorizontalAlignment(Element.ALIGN_CENTER);
    table.addCell(c1);
    table.setHeaderRows(<span class="hl-number">1</span>);

    table.addCell(<span class="hl-string" style="color:blue">"1.0"</span>);
    table.addCell(<span class="hl-string" style="color:blue">"1.1"</span>);
    table.addCell(<span class="hl-string" style="color:blue">"1.2"</span>);
    table.addCell(<span class="hl-string" style="color:blue">"2.1"</span>);
    table.addCell(<span class="hl-string" style="color:blue">"2.2"</span>);
    table.addCell(<span class="hl-string" style="color:blue">"2.3"</span>);

    subCatPart.add(table);

  }

  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> createList(Section subCatPart) {
    List list = <span class="hl-keyword">new</span> List(true, false, <span class="hl-number">10</span>);
    list.add(<span class="hl-keyword">new</span> ListItem(<span class="hl-string" style="color:blue">"First point"</span>));
    list.add(<span class="hl-keyword">new</span> ListItem(<span class="hl-string" style="color:blue">"Second point"</span>));
    list.add(<span class="hl-keyword">new</span> ListItem(<span class="hl-string" style="color:blue">"Third point"</span>));
    subCatPart.add(list);
  }

  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> addEmptyLine(Paragraph paragraph, <span class="hl-keyword">int</span> number) {
    <span class="hl-keyword">for</span> (<span class="hl-keyword">int</span> i = <span class="hl-number">0</span>; i &lt; number; i++) {
      paragraph.add(<span class="hl-keyword">new</span> Paragraph(<span class="hl-string" style="color:blue">" "</span>));
    }
  }
} </pre><p>

                  <div>
                     <h2 class="title"><a name="readpdf"></a>5.&nbsp;Read an existing pdf
                     </h2>
                  </div>
               </div>
            </div>
            <p>
               iText allows to read existing pdf's and include them into your
               own
               pdf. The following example will create page 2 of the previous
               example
               and create a new document with this page. 
               
            </p>
            <p>
               Create a new Java project
               "de.vogella.itext.readpdf" with the package
               "de.vogella.itext.read".
               Create a folder
               "lib" and put the iText library
               (jar file) into
               this
               folder.
               <a class="ulink" href="http://www.vogella.com/tutorials/Eclipse/article.html#classpath" target="_top">
                  Add the jar to your classpath.</a>
               Create the following class "ReadAndUsePdf.java".
               
            </p>
            <p>
               
            </p><pre class="programlisting"><span class="hl-keyword">package</span> de.vogella.itext.read;

<span class="hl-keyword">import</span> java.io.FileOutputStream;
<span class="hl-keyword">import</span> java.io.IOException;

<span class="hl-keyword">import</span> com.itextpdf.text.Document;
<span class="hl-keyword">import</span> com.itextpdf.text.DocumentException;
<span class="hl-keyword">import</span> com.itextpdf.text.Image;
<span class="hl-keyword">import</span> com.itextpdf.text.pdf.PdfImportedPage;
<span class="hl-keyword">import</span> com.itextpdf.text.pdf.PdfReader;
<span class="hl-keyword">import</span> com.itextpdf.text.pdf.PdfWriter;

<span class="hl-keyword">public</span> <span class="hl-keyword">class</span> ReadAndUsePdf {
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> String INPUTFILE = <span class="hl-string" style="color:blue">"c:/temp/FirstPdf.pdf"</span>;
  <span class="hl-keyword">private</span> <span class="hl-keyword">static</span> String OUTPUTFILE = <span class="hl-string" style="color:blue">"c:/temp/ReadPdf.pdf"</span>;

  <span class="hl-keyword">public</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> main(String[] args) <span class="hl-keyword">throws</span> DocumentException,
      IOException {
    Document document = <span class="hl-keyword">new</span> Document();

    PdfWriter writer = PdfWriter.getInstance(document,
        <span class="hl-keyword">new</span> FileOutputStream(OUTPUTFILE));
    document.open();
    PdfReader reader = <span class="hl-keyword">new</span> PdfReader(INPUTFILE);
    <span class="hl-keyword">int</span> n = reader.getNumberOfPages();
    PdfImportedPage page;
    <em class="hl-comment" style="color: #080">// Go through all pages</em>
    <span class="hl-keyword">for</span> (<span class="hl-keyword">int</span> i = <span class="hl-number">1</span>; i &lt;= n; i++) {
      <em class="hl-comment" style="color: #080">// only page number 2 will be included</em>
      <span class="hl-keyword">if</span> (i == <span class="hl-number">2</span>) {
        page = writer.getImportedPage(reader, i);
        Image instance = Image.getInstance(page);
        document.add(instance);
      }
    }
    document.close();

  }

} </pre><p>
