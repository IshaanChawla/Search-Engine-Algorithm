    var subcatFilter = false;
    var specialityFilter = false;
    
	//compares the Categories/subcategories/specialities and arrange them in alphabetical order 
	var alphabetically = function(obj1,obj2){		
        var obj1LowerCase = new String(obj1.toLowerCase());
        var obj2LowerCase = new String(obj2.toLowerCase());
        return (obj1LowerCase.localeCompare(obj2LowerCase));
    }
	
	//set Category/subCategory/speciality in the URL on click of the Category/subCategory/speciality present in the first/second/third Coloumn
	function updateParams(value,level){
		value = value.toLowerCase().replace(/[_\W]+/g, "");
		var currentURL = window.location.hash;
		
		if(level ==1){return window.location.hash ='#cat=' + value;}
		
		else if (level ==2){var position =currentURL.indexOf('&subcat=');
					        var placeholder = '&subcat=' }
		else{var position =currentURL.indexOf('&spec=');
			 var placeholder = '&spec='}
			 
		if (position != -1){currentURL = currentURL.substr(0,position);}
		window.location.hash = currentURL + placeholder + value;
	}

	// filters the courses according to selected speciality in 3rd column
    function filterBySpeciality(courses,subCat,speciality){
        var filteredcourses=[];
        $.each(courses,function(key,value){
            thisCourse = this;
            topics = this.topics;
                $.each(topics,function(index){
                    if(this.subCat == subCat && this.speciality == speciality){
                        filteredcourses[filteredcourses.length] = thisCourse;
                        return;
                    }
                });
        });
        return filteredcourses;
    }
	
    //filters the courses according to selected subcategory in 2nd column
	function filterBySubCat(courses,subCat){
        var filteredcourses=[];
        $.each(courses,function(key,value){
            thisCourse = this;
            topics = this.topics;
                $.each(topics,function(index){
                    if(this.subCat == subCat){
                        filteredcourses[filteredcourses.length] = thisCourse;
                        return;
                    }
                });
        });
        return filteredcourses;
    }
	
    //does all filtering based on selected filters,subcat and speciality
    function getFilteredCourses(allCourses){
        var filteredcourses=allCourses;
        var enabledFilters = [];
        var allDisabled=true;
        for(var filter in filters){
            if(filters[filter]){ allDisabled=false; break;}
        }
        if(allDisabled && subcatFilter == false && specialityFilter == false) return allCourses;
        for(var filter in filters){
            if(filters[filter]){
                enabledFilters[enabledFilters.length]=filter;
            }
        }
        if(subcatFilter!=false)
            filteredcourses = filterBySubCat(filteredcourses,subcatFilter);
        if(specialityFilter!=false)
            filteredcourses = filterBySpeciality(filteredcourses,subcatFilter,specialityFilter);
        if(!allDisabled)
            filteredcourses=filterCourses(filteredcourses,enabledFilters);
        return filteredcourses;
    }

	// selects the speciality in 3rd column 
    function selectSpeciality(obj,speciality){	
		var br3Result = $(".scroller",$("#br3"));
        highlightSelectedTag(obj,3);
        specialityFilter = speciality;
		var position =$(br3Result).scrollTop();
		if (position==0){$(br3Result).scrollTop(specialityPosition);}

	// selects the sub catogries in 2nd column 
	function selectSubCat(obj,subCat){
        var courseResults = $("#courseList"); 
        var uniqueSpeciality={}; 
		var br2Result = $(".scroller",$("#br2"));
        var br3Result = $(".scroller",$("#br3"));
        highlightSelectedTag(obj,2);       
        subcatFilter = subCat;
        specialityFilter = false;
        $.each(currentResults,function(key,value){
                topics = this.topics;
                $.each(topics,function(index){
                    if(this.subCat == subCat)
                        if(this.speciality!='')
                            if (uniqueSpeciality[this.speciality] != true)
                                uniqueSpeciality[this.speciality] = true;
                });	
        });	
        var temp = [];
		var spec =[];
        for(value in uniqueSpeciality){
        temp += value +"/";
        }
        if (temp != ''){
        spec = temp.split("/");
        spec.sort(alphabetically);
        spec.shift();
        }
		var i=0;
        for(var k =0;k<spec.length;k++){
			
			if(i%2==0){
		
            $('<p class="odd" onClick="updateParams(\''+spec[k]+'\',3)">'+spec[k]+'</p>').appendTo($(br3Result));
			}
			else{
		
            $('<p class="even" onClick="updateParams(\''+spec[k]+'\',3)">'+spec[k]+'</p>').appendTo($(br3Result));
			}
			
			i++;
        }
		if(i==0){
		$('<p class="none">No specialties are available within this sub-topic</p>').appendTo($(br3Result));
		}
		var position =$(br2Result).scrollTop();
		if (position ==0){$(br2Result).scrollTop(subCategoryPosition);}		 
        $("#br3").scrollTop(0);
        $(br3Result).scrollTop(0);        
        showFilteredCourses();
    }

	function populateSubCat(){
		var uniqueSubCat={}; 
        var br1Result = $(".scroller",$("#br1"));   
		var br2Result = $(".scroller",$("#br2"));  
        $.each(currentResults,function(key,value){
                topics = this.topics;
                $.each(topics,function(index){
                    if(this.subCat!='')
                        if (uniqueSubCat[this.subCat] != true)
                            uniqueSubCat[this.subCat] = true;
                });
        });		
		var temp = [];
		var subCat = [];
		for(value in uniqueSubCat){
		temp += value +"/";
		}
		subCat = temp.split("/");
		subCat.sort(alphabetically);
		subCat.shift();		
		var j=0;		
        for(var i =0;i<subCat.length;i++){
			if(j%2==0){
            $('<p class="odd" onClick="updateParams(\''+subCat[i].replace("'","\\'")+'\',2)">'+subCat[i]+'</p>').appendTo($(br2Result));
			}
			else{
			$(''+j+'<p class="even" onClick="updateParams(\''+subCat[i].replace("'","\\'")+'\',2)">'+subCat[i]+'</p>').appendTo($(br2Result));
			}
			j++;
        }
		var position =$(br1Result).scrollTop();
		if (position ==0){$(br1Result).scrollTop(categoryPosition);}
        $("#br2").scrollTop(0);
        $(br2Result).scrollTop(0);
        showFilteredCourses();
    }
	
	//select the Specility tag on which user has clicked  
	function selectSpecialityTag(speciality){
		speciality = speciality.replace('spec=','');
		var Br3pTags =$(".scroller",$("#br3")).find("p");		
			$.each(Br3pTags,function(){
				if (speciality==this.innerHTML.toLowerCase().replace(/[_\W]+/g, "")){
				specialityPosition =this.offsetTop - 6*26;   //26 is the height of one row and each box shows 6 rows at a time.
				selectSpeciality(this,this.innerHTML);
				}
				});				
		}
	
	//select the Subcategory tag on which user has clicked   
	function selectSubCategoryTag(subCategory){
		var Br2pTags =$(".scroller",$("#br2")).find("p");
		var SubCat = subCategory.replace('subcat=','');
			$.each(Br2pTags,function(){						
				if (SubCat==this.innerHTML.toLowerCase().replace(/[_\W]+/g, "")){
				subCategoryPosition= this.offsetTop - 6*26;    //26 is the height of one row and each box shows 6 rows at a time.
				selectSubCat(this,this.innerHTML);				
				}
			});	
		}
	
	//select the Category tag on which user has clicked
	function selectCategoryTag(category){
		var Br1pTags = $(".scroller",$("#br1")).find("p");
			$.each(Br1pTags,function(){	
				var rel_category= $(this).attr('rel');				
				if (rel_category == category){
						var file = category.replace('#cat=','')+'.json'	
						selectTag(this,file,1);
						categoryPosition= this.offsetTop - 6*26;   //26 is the height of one row and each box shows 6 rows at a time.
						}
				});
			}
	
	//any change in URL will call this function
	$(window).hashchange(function(){
		var category = location.hash.split('&')[0];
		var subCategory = location.hash.split('&')[1];
		var speciality = location.hash.split('&')[2];	
		
		if (category != undefined && subCategory==undefined){selectCategoryTag(category);
                                                             trackEvents(category,'Category Click','Find By Topic');}	
		
		if (subCategory != undefined && speciality == undefined){
			var Br2pTags =$(".scroller",$("#br2")).find("p");
			if (Br2pTags.length>0){selectSubCategoryTag(subCategory);
                                   trackEvents(category+'&'+subCategory,'SubCategory Click','Find By Topic');}	
			else{
				selectCategoryTag(category);
				$(document).ajaxSuccess(function() {selectSubCategoryTag(subCategory);
				                                     trackEvents(category+'&'+subCategory,'SubCategory Click','Find By Topic');});					
				}
		}
		
		if (speciality != undefined){
			var Br3pTags =$(".scroller",$("#br3")).find("p");
			if (Br3pTags.length>0){selectSpecialityTag(speciality);
			                       trackEvents(category+'&'+subCategory+'&'+speciality,'Speciality Click','Find By Topic');}
			else{
				selectCategoryTag(category);
				$(document).ajaxSuccess(function() {selectSubCategoryTag(subCategory);
													selectSpecialityTag(speciality);
													trackEvents(category+'&'+subCategory+'&'+speciality,'Speciality Click','Find By Topic');
													});					
				}				
		}				
					
	});
