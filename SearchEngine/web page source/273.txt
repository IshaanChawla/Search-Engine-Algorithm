<script type="text/javascript">
$(document).ready(function(){
   var re =  /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;   
   $(document).keypress(function(event) {
    if (event.which == 13) {
        event.preventDefault();
        $('#submitSignUpTutor').trigger('click');
    }
   });
   $('#submitSignUpTutor').click(function(){				
      var uemail = $('#txtSignUpEmailTutor').val();
      var upwd = $('#txtSignUpPwdTutor').val();								
      var ufname = $('#txtSignUpNameTutor').val();								
      var ulname = $('#txtSignUpLastNameTutor').val();	
      if (ufname == '')
      {				
         $('#errDivTutor').html('Please enter first name');
         $('#txtSignUpNameTutor').focus();
         return false;
      }
      else if (uemail == '' || !re.test(uemail))
      {				
         $('#errDivTutor').html('Please enter a valid email id');
         $('#txtSignUpEmailTutor').focus();
         return false;
      }
      else if (upwd == '')
      {			
         $('#errDivTutor').html('Please enter password');
         $('#txtSignUpPwdTutor').focus();
         return false;
      } else if(!$("#chkTerms").is(':checked')){
         $('#errDivTutor').html('');
         alert('Please accept terms and conditions.');
         $('#chkTerms').focus();
         return false;
      }				

      $.ajax({
        url: "absoluteclasses/ajax/ajaxSignUp.php",
        method:"POST",
        data: {"uemail":uemail,"upwd":upwd,"urole":4,"upfnm":ufname,"uplnm":ulname}
      }).done(function(msg) {				 
         if(msg == -1)
         {
            alert("Something went wrong. Please try again later.");
         }
         else if(msg == -2)
         {
            $('.wrapLoader').hide();
            $('#ulSignup').show();
            $('#txtSignUpEmailTutor').focus();					
            $('#errDivTutor').html('The email id you entered is already exist');	
         }
         else
         {	
            $('#errDivTutor').html('');
            $('#ulSignup').hide();
            $('#ancUname').html(msg);
            $('#ulUsername').show();					
            $('.wrapLoader').show();
            setTimeout(function(){
               $(location).attr('href',"absoluteclasses/viewallcourses.htm");	
            }, 3000);					
         }
      });
   }); 
});	
</script>
