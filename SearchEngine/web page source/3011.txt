<div><p><strong><a name="read"></a>Write a C program to read name and marks of n number of students from user and store them in a file</strong></p>

<pre>
<code>
#include &lt;stdio.h&gt;
int main(){
   char name[50];
   int marks,i,n;
   printf(&quot;Enter number of students: &quot;);
   scanf(&quot;%d&quot;,&amp;n);
   FILE *fptr;
   fptr=(fopen(&quot;C:\\student.txt&quot;,&quot;w&quot;));
   if(fptr==NULL){
       printf(&quot;Error!&quot;);
       exit(1);
   }
   for(i=0;i&lt;n;++i)
   {
      printf(&quot;For student%d\nEnter name: &quot;,i+1);
      scanf(&quot;%s&quot;,name);
      printf(&quot;Enter marks: &quot;);
      scanf(&quot;%d&quot;,&amp;marks);
      fprintf(fptr,&quot;\nName: %s \nMarks=%d \n&quot;,name,marks);
   }
   fclose(fptr);
   return 0;
}
</code></pre>
</div>
<div><p><strong><a name="append"></a>Write a C program to read name and marks of n number of students from user and store them in a file. If the file previously exits, add the information of n students.</strong></p>

<pre>
<code>
#include &lt;stdio.h&gt;
int main(){
   char name[50];
   int marks,i,n;
   printf(&quot;Enter number of students: &quot;);
   scanf(&quot;%d&quot;,&amp;n);
   FILE *fptr;
   fptr=(fopen(&quot;C:\\student.txt&quot;,&quot;a&quot;));
   if(fptr==NULL){
       printf(&quot;Error!&quot;);
       exit(1);
   }
   for(i=0;i&lt;n;++i)
   {
      printf(&quot;For student%d\nEnter name: &quot;,i+1);
      scanf(&quot;%s&quot;,name);
      printf(&quot;Enter marks: &quot;);
      scanf(&quot;%d&quot;,&amp;marks);
      fprintf(fptr,&quot;\nName: %s \nMarks=%d \n&quot;,name,marks);
   }
   fclose(fptr);
   return 0;
}
</code></pre>
</div>
<div><p><strong><a name="fwrite"></a>Write a C program to write all the members of an array of strcures to a file using fwrite(). Read the array from the file and display on the screen.</strong></p>

<pre>
<code>
#include &lt;stdio.h&gt;
struct s
{
char name[50];
int height;
};
int main(){
    struct s a[5],b[5];   
    FILE *fptr;
    int i;
    fptr=fopen(&quot;file.txt&quot;,&quot;wb&quot;);
    for(i=0;i&lt;5;++i)
    {
        fflush(stdin);
        printf(&quot;Enter name: &quot;);
        gets(a[i].name);
        printf(&quot;Enter height: &quot;); 
        scanf(&quot;%d&quot;,&amp;a[i].height); 
    }
    fwrite(a,sizeof(a),1,fptr);
    fclose(fptr);
    fptr=fopen(&quot;file.txt&quot;,&quot;rb&quot;);
    fread(b,sizeof(b),1,fptr);
    for(i=0;i&lt;5;++i)
    {
        printf(&quot;Name: %s\nHeight: %d&quot;,b[i].name,b[i].height);
    }
    fclose(fptr);
}</code></pre>
