                  <div>
                     <h2 class="title"><a name="overview"></a>1.&nbsp;Overview
                     </h2>
                  </div>
               </div>
            </div>
            <p> The Java language provides direct support for time-based
               objects. This article gives a few examples how this API can be used.
               
            </p>
            <p>
               The
               <code class="code">java.util.Date</code>
               and the
               <code class="code">java.util.Calendar</code>
               classes provide
               access to storing
               and
               manipulating dates.
               
            </p>
            <p>
               It is recommended to use
               <code class="code">Calendar</code>
               if possible. Existing API may required that
               you convert from
               <code class="code">Date</code>
               to
               <code class="code">Calendar</code>
               and vice versa.
               
            </p>
         </div>

               <code class="code">SimpleDateFormat</code>
               class. The following snippet gives several example for its usage.
               
            </p>
            <p>
               
            </p><pre class="programlisting"><em class="hl-comment" style="color: #080">// use dd/MM/yy as format</em>

DateFormat df1 = <span class="hl-keyword">new</span> SimpleDateFormat(<span class="hl-string" style="color:blue">"dd/MM/yy"</span>);
String formattedDate1 = df1.format(<span class="hl-keyword">new</span> Date());

<em class="hl-comment" style="color: #080">// or use yyyy/MM/dd as format</em>

DateFormat df2 = <span class="hl-keyword">new</span> SimpleDateFormat(<span class="hl-string" style="color:blue">"yyyy/MM/dd"</span>);
String formattedDate2 = df2.format(theDate); </pre><p>

                        </h3>
                     </div>
                  </div>
               </div>
               <p>
                  The
                  <code class="code">java.util.Calendar</code>
                  class is an abstract encapsulation of
                  the
                  <code class="code">Date</code>
                  object.
                  
               </p>

               <p> Use the following commands to convert to a Date from various
                  formats.
                  
               </p>
               <p>
                  
               </p><pre class="programlisting"><span class="hl-keyword">package</span> conversion;

<span class="hl-keyword">import</span> java.text.ParseException;
<span class="hl-keyword">import</span> java.text.SimpleDateFormat;
<span class="hl-keyword">import</span> java.util.Calendar;
<span class="hl-keyword">import</span> java.util.Date;
<span class="hl-keyword">import</span> java.util.GregorianCalendar;

<span class="hl-keyword">public</span> <span class="hl-keyword">class</span> ConversionExamplesDate {

  <em class="hl-comment" style="color: #080">// convert from String to date</em>
  <span class="hl-keyword">private</span> <span class="hl-keyword">void</span> stringToDate() {
    
    <span class="hl-keyword">try</span> {
      Date date1;
      date1 = <span class="hl-keyword">new</span> SimpleDateFormat(<span class="hl-string" style="color:blue">"MM/dd/yy"</span>).parse(<span class="hl-string" style="color:blue">"05/18/05"</span>);
      System.out.println(date1);
      Date date2 = <span class="hl-keyword">new</span> SimpleDateFormat(<span class="hl-string" style="color:blue">"MM/dd/yyyy"</span>).parse(<span class="hl-string" style="color:blue">"05/18/2007"</span>);
      System.out.println(date2);
    } <span class="hl-keyword">catch</span> (ParseException e) {
      e.printStackTrace();
    }
  }

  <em class="hl-comment" style="color: #080">// convert from millisecs to a String with a defined format</em>
  <span class="hl-keyword">private</span> <span class="hl-keyword">void</span> calcDate(<span class="hl-keyword">long</span> millisecs) {
    SimpleDateFormat date_format = <span class="hl-keyword">new</span> SimpleDateFormat(<span class="hl-string" style="color:blue">"MMM dd,yyyy HH:mm"</span>);
    Date resultdate = <span class="hl-keyword">new</span> Date(millisecs);
    System.out.println(date_format.format(resultdate));
  }
  
  <span class="hl-keyword">private</span> <span class="hl-keyword">void</span> writeActualDate(){
    Calendar cal = <span class="hl-keyword">new</span> GregorianCalendar();
    Date creationDate = cal.getTime();
    SimpleDateFormat date_format = <span class="hl-keyword">new</span> SimpleDateFormat(<span class="hl-string" style="color:blue">"MMM dd,yyyy HH:mm"</span>);
    System.out.println(date_format.format(creationDate));
  }
  

  <span class="hl-keyword">public</span> <span class="hl-keyword">static</span> <span class="hl-keyword">void</span> main(String[] args) {
    ConversionExamplesDate convert = <span class="hl-keyword">new</span> ConversionExamplesDate();
    convert.stringToDate();
    convert.calcDate(System.currentTimeMillis());
    convert.writeActualDate();
  }
